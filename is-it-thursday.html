Uses Python in-browser via PyScript.
from js import Date, document, Intl # Use the visitor's local time (from the browser) today = Date() is_thursday = today.getDay() == 4 # 0=Sun … 4=Thu msg = "It's Thursday" if is_thursday else "It's not Thursday" document.getElementById("answer").textContent = msg # Nicety: show their local date for transparency formatter = Intl.DateTimeFormat( None, { "weekday": "long", "year": "numeric", "month": "long", "day": "numeric" } ) document.getElementById("detail").textContent = f"Your local date is {formatter.format(today)}."
